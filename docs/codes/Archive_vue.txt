const activeId = ref('pepper')

const heatChartEl = ref(null)

function formatCoord(coord) {
  return `${coord[0].toFixed(4)}E, ${coord[1].toFixed(4)}N`
}

function ensureCharts() {
  if (climateChartEl.value && !climateChart) climateChart = echarts.init(climateChartEl.value, null, { renderer: 'canvas' })

  if (standardRadarEl.value && !standardRadar) standardRadar = echarts.init(standardRadarEl.value, null, { renderer: 'canvas' })
}

function renderCharts() {
  ensureCharts()
  climateChart?.setOption(climateOption(), true)
  heatChart?.setOption(heatOption(), true)
  standardRadar?.setOption(benchmarkOption(), true)
  resizeCharts()
}

  climateChart?.resize()
  heatChart?.resize()
  standardRadar?.resize()
}

function initTerrainMap() {
  if (!terrainMapEl.value || terrainMap) return

  terrainMap = new maplibregl.Map({
    container: terrainMapEl.value,
    style: MAP_STYLE,
    center: dossier.value.originPoint.coord,
    zoom: 6.2,
    pitch: 0,
    bearing: 0,
    antialias: true,
    attributionControl: false,
    interactive: false,
  })

  terrainMap.on('load', () => {
    mapLoaded = true
    terrainMap.addSource('active-province', {
      type: 'geojson',
      data: buildProvinceFeatureCollection(),
    })
    terrainMap.addLayer({
      id: 'active-province-fill',
      type: 'fill',
      source: 'active-province',
      paint: {
        'fill-color': dossier.value.color,
        'fill-opacity': 0.16,
      },
    })
    terrainMap.addLayer({
      id: 'active-province-line',
      type: 'line',
      source: 'active-province',
      paint: {
        'line-color': dossier.value.color,
        'line-width': 2,
        'line-opacity': 0.9,
      },
    })
    updateTerrainMap()
  })
}

function buildProvinceFeatureCollection() {
  return {
    type: 'FeatureCollection',
    features: provinceFeature.value ? [provinceFeature.value] : [],
  }
}

function updateTerrainMap() {
  if (!terrainMap || !mapLoaded) return

  const source = terrainMap.getSource('active-province')
  source?.setData(buildProvinceFeatureCollection())

  if (terrainMap.getLayer('active-province-fill')) {
    terrainMap.setPaintProperty('active-province-fill', 'fill-color', dossier.value.color)
    terrainMap.setPaintProperty('active-province-line', 'line-color', dossier.value.color)
  }

  syncOriginMarker()
  fitProvinceBounds()
}

function syncOriginMarker() {
  if (!terrainMap) return
  const markerEl = document.createElement('div')
  markerEl.className = 'origin-marker'
  markerEl.innerHTML = `<span></span><strong>${dossier.value.name}</strong>`

  originMarker?.remove()
  originMarker = new maplibregl.Marker({ element: markerEl, anchor: 'bottom' })
    .setLngLat(dossier.value.originPoint.coord)
    .addTo(terrainMap)
}

function fitProvinceBounds() {
  const bounds = getFeatureBounds(provinceFeature.value)
  if (!bounds) {
    terrainMap.jumpTo({ center: dossier.value.originPoint.coord, zoom: 6.2 })
    return
  }
  terrainMap.fitBounds(bounds, { padding: 28, duration: 0, maxZoom: 6.6 })
}

function getFeatureBounds(feature) {
  if (!feature?.geometry?.coordinates) return null
  const bounds = new maplibregl.LngLatBounds()
  walkCoordinates(feature.geometry.coordinates, point => bounds.extend(point))
  return bounds.isEmpty() ? null : bounds
}

function walkCoordinates(value, visit) {
  if (!Array.isArray(value)) return
  if (typeof value[0] === 'number' && typeof value[1] === 'number') {
    visit(value)
    return
  }
  value.forEach(item => walkCoordinates(item, visit))
}

onMounted(async () => {
  await nextTick()
  renderCharts()
  initTerrainMap()
  window.addEventListener('resize', resizeCharts)
})

watch(activeId, async () => {
  await nextTick()
  renderCharts()
  updateTerrainMap()
})

onUnmounted(() => {
  window.removeEventListener('resize', resizeCharts)
  climateChart?.dispose()
  heatChart?.dispose()
  standardRadar?.dispose()
  originMarker?.remove()
  terrainMap?.remove()
})
</script>
